**Documentos.jsx migration:**

```javascript
import { useQuery, useQueryClient } from '@tanstack/react-query';
import { useAeroportos } from '@/components/lib/useStaticData';

// Inside component — replace loadData:
const { user: currentUser } = useAuth();
const empresaId = currentUser?.empresa_id;
const queryClient = useQueryClient();

const { data: aeroportosRaw = [] } = useAeroportos();
const aeroportos = useMemo(() => {
  const angola = aeroportosRaw.filter(a => a.pais === 'AO');
  return getAeroportosPermitidos(currentUser, angola, currentUser?.empresa_id);
}, [aeroportosRaw, currentUser]);

const { data: documentos = [], isLoading } = useQuery({
  queryKey: ['documentos', empresaId],
  queryFn: async () => {
    const docs = empresaId
      ? await Documento.filter({ empresa_id: empresaId }, '-data_publicacao')
      : await Documento.list('-data_publicacao');
    const angola = aeroportosRaw.filter(a => a.pais === 'AO');
    return filtrarDadosPorAcesso(currentUser, docs, 'aeroporto', angola);
  },
  staleTime: 0,
  refetchOnWindowFocus: false,
  enabled: !!currentUser,
});

// At all mutation sites (create, update, delete, archive), replace loadData() with:
queryClient.invalidateQueries({ queryKey: ['documentos', empresaId] });
```

Remove: loadData function definition, the useEffect that called loadData, useState for documentos and aeroportos (these come from hooks now). Keep any other local UI state (modals, selected items, etc.).

Proforma is the most complex of the three — it has create, update, cancel, and delete mutation sites. Apply the same migration pattern carefully.

```javascript
import { useQuery, useQueryClient } from '@tanstack/react-query';
import { useAeroportos } from '@/components/lib/useStaticData';

// Inside component:
const { effectiveEmpresaId } = useCompanyView();
const { user: authUser } = useAuth();
const currentUser = ensureUserProfilesExist(authUser); // keep this — not fetch-related
const empresaId = effectiveEmpresaId || currentUser?.empresa_id;
const queryClient = useQueryClient();

const { data: aeroportosRaw = [] } = useAeroportos();
const aeroportos = useMemo(() => {
  return getAeroportosPermitidos(currentUser, aeroportosRaw.filter(a => a.pais === 'AO'), effectiveEmpresaId);
}, [aeroportosRaw, currentUser, effectiveEmpresaId]);

const { data: proformas = [], isLoading } = useQuery({
  queryKey: ['proformas', empresaId],
  queryFn: () => {
    const filters = {};
    if (empresaId) filters.empresa_id = empresaId;
    return Proforma.filter(filters, '-created_date');
  },
  staleTime: 0,
  refetchOnWindowFocus: false,
  enabled: !!currentUser,
});

// At ALL mutation sites (create, update, cancel, delete proforma items, etc.):
// Replace: await loadData()  with:
await queryClient.invalidateQueries({ queryKey: ['proformas', empresaId] });
```

Important: Some mutations use `await loadData()` to ensure fresh data before continuing. Replace with `await queryClient.invalidateQueries(...)` which returns a Promise — awaiting it ensures the refetch is initiated before continuing.
