Create src/hooks/useEntityQuery.js with a factory function makeEntityQuery. This is a thin wrapper that standardizes how remaining pages create useQuery-powered hooks without each page repeating boilerplate.

```javascript
import { useQuery } from '@tanstack/react-query';

/**
 * Factory that creates a useQuery-powered data hook for a page-specific entity fetch.
 *
 * @param {string[]} queryKey - TanStack Query key array. MUST include empresaId as second element.
 * @param {Function} queryFn  - Async function that returns the data array.
 * @param {Object}   options  - Optional overrides merged into useQuery options.
 * @returns Object with { data, isLoading, isError, error, refetch }
 *
 * Usage in a page:
 *   const { data: movimentos = [], isLoading } = makeEntityQuery(
 *     ['movimentos', empresaId],
 *     () => MovimentoFinanceiro.filter({ empresa_id: empresaId }, '-data'),
 *     { enabled: !!empresaId }
 *   )();
 *
 * Or create a named hook:
 *   const useMovimentos = (empresaId) => makeEntityQuery(
 *     ['movimentos', empresaId],
 *     () => MovimentoFinanceiro.filter({ empresa_id: empresaId }, '-data'),
 *     { enabled: !!empresaId }
 *   );
 */
export function makeEntityQuery(queryKey, queryFn, options = {}) {
  return function useEntityData() {
    return useQuery({
      queryKey,
      queryFn,
      staleTime: 0,           // Operational data — always fresh on mount
      gcTime: 1000 * 60 * 5,  // Keep in memory 5 min after unmount
      refetchOnWindowFocus: false,
      ...options,
    });
  };
}
```

staleTime: 0 is correct for operational pages (movimentos, ocorrencias, inspecoes) — these are not reference data. The queryClientInstance default staleTime of 2 min is overridden here.

**FundoManeio.jsx migration:**

Replace the loadData pattern for MovimentoFinanceiro. FundoManeio has filter params (aeroportoId, dataInicio, dataFim) that affect the query — these must be in the queryKey so changing filters refetches:

```javascript
import { useQuery, useQueryClient } from '@tanstack/react-query';
import { useAeroportos } from '@/components/lib/useStaticData';

// Inside component:
const { effectiveEmpresaId } = useCompanyView();
const { user: currentUser } = useAuth();
const empresaId = effectiveEmpresaId || currentUser?.empresa_id;
const queryClient = useQueryClient();

// Replace loadData aeroporto fetch with:
const { data: aeroportosRaw = [] } = useAeroportos();
// Apply existing filtering logic (getAeroportosPermitidos, pais==='AO') on aeroportosRaw in useMemo

// Replace loadData movimentos fetch with:
const { data: movimentos = [], isLoading } = useQuery({
  queryKey: ['movimentos', empresaId, aeroportoId, dataInicio, dataFim],
  queryFn: async () => {
    const query = { empresa_id: empresaId };
    if (aeroportoId && aeroportoId !== 'todos') query.aeroporto_id = aeroportoId;
    if (dataInicio) query.data = { $gte: dataInicio };
    if (dataFim) query.data = { ...query.data, $lte: dataFim };
    const data = await MovimentoFinanceiro.filter(query, '-data');
    return data.filter(m => userAccessibleAirportIds.includes(m.aeroporto_id));
  },
  enabled: !!empresaId,
  staleTime: 0,
  refetchOnWindowFocus: false,
});

// At every mutation site (create, update, delete), replace: loadData() with:
queryClient.invalidateQueries({ queryKey: ['movimentos', empresaId] });
```

Note: userAccessibleAirportIds is derived from aeroportosRaw — ensure this derivation happens in useMemo before the useQuery call, or use initialData pattern if needed. Keep the existing derivation logic; just change where aeroportos come from.

Remove: loadData useCallback, the useEffect that called loadData(), and the useState for aeroportos/empresas that are now served by useAeroportos(). Keep setIsLoading only if there's a loading spinner driven by non-query state.
